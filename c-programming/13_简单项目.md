# 简单项目 

## 猜数字游戏的简单实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void menu(void) {
  printf("**********************************************\n");
  printf("****************    1.Play    ****************\n");
  printf("****************    2.Exit    ****************\n");
  printf("**********************************************\n");
}

void game(void) {
  int guess = 0;
  int ret = rand() % 100 + 1;
  int chance = 0;
  while (1) {
    printf("请猜数字:>");
    if (scanf("%d", &guess) != 1)  // scanf 函数返回值是读取字符数目
    {
      printf("输入错误，请重新输入数字\n");
      // 清空输入缓冲区
      while (getchar() != '\n');
      continue;
    }
    if (guess != ret && chance == 2) {
      printf("游戏失败，重新开始\n");
      break;
    }
    if (guess < ret) {
      printf("猜小了\n");
    } else if (guess > ret) {
      printf("猜大了\n");
    } else {
      printf("恭喜你，猜对了\n");
      break;
    }
    chance++;
    printf("你已猜测 %d  次，还有%d 次机会\n", chance, 3 - chance);
  }
}

int main(void) {
  int input = 0;
  int n = 0;
  srand((unsigned int)time(0));

  do {
    menu();
    printf("请选择(1\\2):>");
    if (scanf("%d", &input) != 1) {
      printf("输入错误，请重新选择\n");
      // 清空输入缓冲区
      while (getchar() != '\n');
      continue;
    }

    switch (input) {
      case 1:
        printf("游戏开始\n");
        printf("(数字范围:1~100)\n");
        game();
        printf("你已进行%d次游戏，适当游戏益脑，过度游戏伤身\n", ++n);
        break;

      case 2:
        printf("退出游戏\n");
        break;

      default:
        printf("选择错误，请重新选择\n");
        break;
    }

    if (n == 10) {
      printf("注意休息\n");
      break;
    }
  } while (input != 2);

  return 0;
}
```

## 三子棋

### `game.c`

```c
#include "Sasuke_game.h"

void InitBoard(char board[ROW][COL], int row, int col) {
  int i = 0;
  int j = 0;

  for (i = 0; i < row; i++) {
    for (j = 0; j < col; j++) {
      board[i][j] = ' ';
    }
  }
}

void DisplayBoard(char board[ROW][COL], int row, int col) {
  int i = 0;

  for (i = 0; i < row; i++) {
    int j = 0;

    for (j = 0; j < col; j++) {
      printf(" %c ", board[i][j]);
      if (j < col - 1) {
        printf("|");
      }
    }
    printf("\n");
    if (i < row - 1) {
      int j = 0;

      for (j = 0; j < col; j++) {
        printf("---");
        if (j < col - 1) printf("|");
      }
      printf("\n");
    }
  }
}

void PlayerMove(char board[ROW][COL], int row, int col) {
  int x = 0;
  int y = 0;

  printf("玩家下棋\n");
  while (1) {
    printf("请输入坐标:>\n");
    scanf("%d %d", &x, &y);
    if (x >= 1 && x <= row && y >= 1 && y <= col) {
      if (board[x - 1][y - 1] == ' ') {
        board[x - 1][y - 1] = '*';
        break;
      } else
        printf("坐标被占用，请选择其他位置\n");
    } else
      printf("坐标违法，请重新输入:>\n");
  }
}

void ComputerMove(char board[ROW][COL], int row, int col) {
  int x = 0;
  int y = 0;

  printf("电脑下棋\n");
  while (1) {
    x = rand() % row;
    y = rand() % col;
    if (board[x][y] == ' ') {
      board[x][y] = '#';
      break;
    }
  }
}

char IsWin(char board[ROW][COL], int row, int col) {
  int i = 0;
  int j = 0;

  for (i = 0; i < row; i++) {
    if (board[i][0] != ' ') {
      for (j = 0; j < col; j++) {
        if (board[i][j] != board[i][0]) break;
      }
      if (j == col) return board[i][0];
    }
  }
  for (j = 0; j < row; j++) {
    if (board[0][j] != ' ') {
      for (i = 0; i < row; i++) {
        if (board[i][j] != board[0][j]) break;
      }
      if (i == row) return board[0][j];
    }
  }
  if (board[0][0] != ' ') {
    for (i = 0, j = 0; i < row && j < col; i++, j++) {
      if (board[i][j] != board[0][0]) break;
    }
    if (i == row && j == col) return board[0][0];
  }
  if (board[0][col - 1] != ' ') {
    for (i = 0, j = 0; i < row && j < col; i++, j++) {
      if (board[i][col - 1 - j] != board[0][col - 1]) break;
    }
    if (i == row && j == col) return board[0][col - 1];
  }
  if (IsFull(board, ROW, COL)) {
    return 'Q';
  }

  return 'C';
}

int IsFull(char board[ROW][COL], int row, int col) {
  int i = 0;
  int j = 0;

  for (i = 0; i < ROW; i++) {
    for (j = 0; j < COL; j++) {
      if (board[i][j] == ' ') return 0;
    }
  }

  return 1;
}
```

```c
#define _CRT_SECURE_NO_WARNINGS
#include "Sasuke_game.h"

  printf("**********************************\n");

  char ret = 0;
  char board[ROW][COL] = {0};

  InitBoard(board, ROW, COL);
  DisplayBoard(board, ROW, COL);
  while (1) {
    PlayerMove(board, ROW, COL);
    ret = IsWin(board, ROW, COL);
    if (ret != 'C') {
      break;

    DisplayBoard(board, ROW, COL);
    ComputerMove(board, ROW, COL);
    ret = IsWin(board, ROW, COL);
    if (ret != 'C') {
      break;
    }
    DisplayBoard(board, ROW, COL);

  if (ret == '*')
    printf("玩家赢了\n");
  else if (ret == '#')
    printf("电脑赢了\n");
  else
    printf("平局\n");
}

  int input = 0;

  srand((unsigned int)time(NULL));
  do {
    menu();
    printf("请选择:>");
    scanf("%d", &input);
    switch (input) {
      case 1:
        game();
        break;
      case 2:
        printf("退出游戏\n");
        break;
      default:
        printf("选择错误\n");
        break;
    }
  } while (input != 2);

  return 0;
}
```

### `game.c`

```c
#define _CRT_SECURE_NO_WARNINGS
#include "game.h"

void init_board(char board[ROWS][COLS], int rows, int cols, char set) {
  int i = 0;
  int j = 0;

  for (i = 0; i < rows; i++) {
    for (j = 0; j < cols; j++) {
      board[i][j] = set;
    }
  }
}

void display_board(char board[ROWS][COLS], int row, int col) {
  int i = 0;
  int j = 0;

  printf("----- 扫雷游戏 -----\n");
  for (j = 0; j <= col; j++)  // 打印每一列的列号
  {
    printf("%d ", j);
  }
  printf("\n");
  for (i = 1; i <= row; i++) {
    printf("%d ", i);  // 打印每一行的行号
    for (j = 1; j <= col; j++) {
      printf("%c ", board[i][j]);  // 打印出每个雷区
    }
    printf("\n");
  }
  printf("----- 扫雷游戏 -----\n");
}

void set_mine(char board[ROWS][COLS], int row, int col) {
  int count = EASY_COUNT;

  while (count) {
    int x = rand() % row + 1;
    int y = rand() % col + 1;

    if (board[x][y] == '0') {
      board[x][y] = '1';
      count--;
    }
  }
}

int get_mine_count(char board[ROWS][COLS], int x, int y) {
  return (board[x - 1][y - 1] + board[x - 1][y] + board[x - 1][y + 1] +
          board[x][y - 1] + board[x][y + 1] + board[x + 1][y - 1] +
          board[x + 1][y] + board[x + 1][y + 1] -
          8 * '0');  // 减去 8 个字符 '0',如 '1' - '0',最后算出数字 1
}

void find_mine(char mine[ROWS][COLS], char show[ROWS][COLS], int row, int col) {
  int x = 0;
  int y = 0;
  int win = 0;  // 找到非雷的个数

  while (win < row * col - EASY_COUNT) {
    printf("请输入要排查的坐标:>\n");
    scanf("%d %d", &x, &y);
    if (x >= 1 && x <= row && y >= 1 && y <= col) {
      if (show[x][y] != '*')
        printf("该坐标被排查过，不能重复排查\n");
      else {
        if (mine[x][y] == '1') {
          printf("抱歉,你被炸死了\n");
          display_board(mine, ROW, COL);
          break;
        } else {
          win++;
          int count =
              get_mine_count(mine, x, y);  // 排查 min e数组中，(x, y) 周围有几个雷
          show[x][y] = count + '0';        // 通过加上字符 '0' 将 count 转换成数字
          display_board(show, ROW, COL);
        }
      }
    } else
      printf("输入坐标非法，请您重新输入\n");
  }
  if (win == row * col - EASY_COUNT) {
    printf("恭喜你,排雷成功\n");
    display_board(mine, ROW, COL);
  }
}
```

```c
#define _CRT_SECURE_NO_WARNINGS
#include "Sasuke_game.h"

void init_board(char board[ROWS][COLS], int rows, int cols, char set) {
  int i = 0;

void set_mine(char board[ROWS][COLS], int row, int col) {
  int count = EASY_COUNT;

  while (count) {
    int x = rand() % row + 1;
    int y = rand() % col + 1;

      board[x][y] = '$';
      count--;
    }
  }

int get_mine_count(char board[ROWS][COLS], int x, int y) {
  return ((board[x - 1][y - 1] + board[x - 1][y] + board[x - 1][y + 1] +
           board[x][y - 1] + board[x][y + 1] + board[x + 1][y - 1] +
           board[x + 1][y] + board[x + 1][y + 1] - 8 * ' ') /
          4);
}

int find_mine(char mine[ROWS][COLS], char show[ROWS][COLS], int row, int col) {
  int x = 0;
  int y = 0;
  int input = 0;
  int win = 0;  // 找到非雷的个数

  while (win < row * col - EASY_COUNT) {
    printf("请输入要操作的坐标:>\n");
    scanf("%d %d", &x, &y);
    if (x >= 1 && x <= row && y >= 1 && y <= col) {
      printf("请输入要执行的操作:>\n1.排查 2.标记 3.取消标记\n");
      scanf("%d", &input);
      switch (input) {
        case 1:
          if (show[x][y] != '*' && show[x][y] != '#')
            printf("该坐标被排查过,不能重复排查\n");
          else if (show[x][y] != '*' && show[x][y] == '#')
            printf("该坐标被标记,如要标记,请先取消标记\n");
          else {
            if (mine[x][y] == '$') {
              printf("抱歉,你被炸死了\n");
              display_board(mine, ROW, COL);
              return 0;  // 通过return来直接跳出整个循环
            } else {
              extend_no_mine(show, mine, x, y);
              win = count_win(show, mine, ROW, COL);
              display_board(show, ROW, COL);
            }
          }
          break;
        case 2:
          if (show[x][y] != '#' && show[x][y] == '*') {
            show[x][y] = '#';
            display_board(show, ROW, COL);
          } else if (show[x][y] == '#')
            printf("该坐标无法标记,请重新输入\n");
          else
            printf("该坐标已被排查，请重新输入\n");
          break;
        case 3:
          if (show[x][y] == '#') {
            show[x][y] = '*';
            display_board(show, ROW, COL);
          } else
            printf("该坐标无标记,请重新输入\n");
          break;
        default:
          printf("输入选项错误,请重新输入\n");
          break;
      }
    } else
      printf("输入坐标非法，请您重新输入\n");
  }
  if (win == row * col - EASY_COUNT) {
    printf("恭喜你,排雷成功\n");
    display_board(mine, ROW, COL);
  }

  return 0;
}

void extend_no_mine(char show[ROWS][COLS], char mine[ROWS][COLS], int x,
                    int y)  // 展开
{
  int i = 0;
  int j = 0;
  int count = get_mine_count(mine, x, y);

  if (mine[x][y] == ' ' && show[x][y] == '*' &&
      count ==
          0)  // 本身不为地雷，周边没有地雷，并且未被排查，如果周边有雷，就要显示雷的个数
  {
    show[x][y] = ' ';
    for (i = x - 1; i < x + 2; i++) {
      for (j = y - 1; j < y + 2; j++) {
        extend_no_mine(show, mine, i, j);
      }
    }
  } else if (show[x][y] == ' ')
    ;  // 表示排查已经被判为0的数,不执行语句(主要为了防止递归的时候，再次循环到)
  else                         // 显示mine不为零的方块
    show[x][y] = count + '0';  // 本身不为地雷，周边存在地雷
}

int count_win(char show[ROWS][COLS], char mine[ROWS][COLS], int row,
              int col)  // 计算win的数量
{
  int i = 0;
  int j = 0;
  int sum = 0;

  for (i = 1; i <= row; i++) {
    for (j = 1; j <= col; j++) {
      if (show[i][j] != '*' &&
          show[i][j] != '#')  // 代表排查出的数量，除了未排查的和标记的
        sum++;
    }
  }

  return sum;
}
```
