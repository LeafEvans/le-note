# 查找与排序功能

## 查找

```cpp
template <class IT, class T>
IT find(const IT &begin, const IT &end, const T &data) {
  for (IT it = begin; it != end; ++it) {
    if (*it == data) {
      return it;
    }
  }
  return end;
}

void print(const string &str, const List<int> &list) {
  cout << str << endl;
  for (const int elem : list) {
    cout << elem << ' ';
  }
  cout << '\n' << "----------------------------------\n";
}

int main() {
  typedef List<int>::Iterator IT;
  List<int> list;

  for (int i = 0; i < 5; ++i) {
    list.push_front(10 + i);
  }
  for (int i = 0; i < 5; ++i) {
    list.push_back(100 + i);
  }
  print("初始化列表后：", list);
  IT fit = find(list.begin(), list.end(), 100);
  if (fit != list.end()) {
    list.erase(fit);
  }
  print("找到 100 并删除后：", list);
  return 0;
}
```

### 利用 `<` 实现排序

```cpp
template <class IT>
void sort(const IT &begin, const IT &end) {
  IT p = begin;   // 指向初始节点
  IT last = end;  // 指向尾部节点
  --last;

  // 当 i != j 时，进行循环
  for (IT i = begin, j = last; i != j;) {
    // 当 i != p and (i) < (p) 时，进行循环
    while (i != p && *i < *p) {
      // 往后寻找
      ++i;
    }
    // 如果 i != p 时，说明是 (i) > (p)
    if (i != p) {
      // 进行交换，更新 p 的值
      swap(*i, *p);
      p = i;
    }

    // 类似的
    while (j != p && *p < *j) {
      --j;
    }
    if (j != p) {
      swap(*j, *p);
      p = j;
    }
  }

  IT it = begin;
  ++it;
  // it 为头部节点的下一个元素
  // p 不等于 begin and 不等于 it
  // 说明 p 指向的节点左边至少有 2 个元素
  if (p != begin && p != it) {
    sort(begin, p);
  }

  it = p;
  ++it;
  // it 不等于终止迭代，也不等于尾节点
  // 而 it 等于 p + 1
  // 因此 p 指向的节点右边至少有 2 个元素
  if (it != end && it != last) {
    sort(it, end);
  }
}
```

### 利用比较器实现排序

需要在比较器类中重载 `()` 操作符。

```cpp
template <class IT, class CMP>
void sort(const IT &begin, const IT &end, CMP &cmp) {
  IT p = begin;
  IT last = end;
  --last;

  for (IT i = begin, j = last; i != j;) {
    while (i != p && cmp(*i, *p)) {
      ++i;
    }
    if (i != p) {
      swap(*i, *p);
      p = i;
    }

    while (j != p && cmp(*p, *j)) {
      --j;
    }
    if (j != p) {
      swap(*j, *p);
      p = j;
    }

    IT it = begin;
    ++it;
    if (p != it && p != begin) {
      sort(begin, p, cmp);
    }

    it = p;
    ++it;
    if (it != end && it != last) {
      sort(it, last, cmp);
    }
  }
}

class CMP {
 public:
  bool operator()(const int &a, const int &b) { return a < b; }
};
```

```cpp
int main() {
  typedef List<int>::Iterator IT;
  List<int> list;

  for (int i = 0; i < 5; ++i) {
    list.push_front(10 + i);
  }
  for (int i = 0; i < 5; ++i) {
    list.push_back(100 + i);
  }
  print("初始化列表后：", list);

  IT fit = find(list.begin(), list.end(), 100);
  if (fit != list.end()) {
    list.erase(fit);
  }
  print("找到 100 并删除后：", list);

  CMP cmp;
  sort(list.begin(), list.end(), cmp);
  print("排序后：", list);

  return 0;
}
```
